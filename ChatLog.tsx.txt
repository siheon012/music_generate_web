import React from 'react';

interface Message {
  id: number;
  text: string;
  isUser: boolean;
}

interface ChatLogProps {
  messages: Message[];
}

const ChatLog: React.FC<ChatLogProps> = ({ messages }) => {
  return (
    <div className="chat-messages">
      {messages.map((message) => (
        <div
          key={message.id}
          className={`message ${
            message.isUser ? 'message-user' : 'message-ai'
          }`}
        >
          {message.text}
        </div>
      ))}
    </div>
  );
};

export default ChatLog;
